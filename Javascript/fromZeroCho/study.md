    함수를 호출하면 ? 
    만약 아래같은 코드가 있다고 치자.

  ```js
    function c() {
      const y = 'y';
      console.log('c');
    }

    function a () {
      const x = 'x';
      console.log('a');
      function b() {
        const z = 'z';
        console.log('b');
        c();
      }
      b();
    }
    
    a();
    c();
  ```
    위에서 a를 호출했다. 
    함수의 호출은 그 함수의 선언부로가서 => 
    이렇게 실행시킴.
    그래서 a()를 호출후에 c()가 호출되는것이 아니고!! 
    a의 선언부로 가서 순서대로 실행을 시키는거다. 그 후에 c()가 호출됌!!!
